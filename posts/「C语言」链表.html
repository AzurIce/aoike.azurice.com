链表

一、单向链表

节点(Node) 结构体：

struct Node {
    int id;
	Node *next;
	/* ... */
}

由于链表的性质，只需要访问到第一个节点，就可以访问到后续所有节点，因此常使用 一个头指针 或 一个头结点 来维护链表。

头指针：

Node *head = NULL;

头结点：

Node head;
head.next = NULL;

1. 添加节点

  1. 创建节点

    这一步是在内存中开辟一段空间给节点，并把相应数据存储进去。

    Node *createNode(int id) {
	Node *p = malloc(sizeof(Node));
    p -> id = id;
    return p;
}

  2. 将节点添加到链表中

    头指针：

    void *addNode(Node *p, Node **head) {
    p->next = *head;
    *head = p;
}

    头结点：

    void addNode(Node *p, Node *head) {
    p->next = head->next;
	head->next = p;
}

2. 删除节点

删除节点p

将节点p从链表中移除并释放内存

头指针：

// 删除头结点
void delHeadNode(Node **head) {
    if (*head == NULL) return;
    Node *del = *head;
    
    *head = del->next;
    free(del);
}
// 删除 p 节点后的一个节点
void delNode(Node *p) {
    if (p->next == NULL) return;
    Node *del = p->next;
    
    p->next = del->next;
    free(del);
}

头结点：

void delNode(Node *p) {
    if (p->next == NULL) return;
    Node *del = p->next;
    
    p->next = del->next;
    free(del);
}

3. 遍历链表

头指针：

void printList(Node *head) {
    Node *i;
    for (i = head; i != NULL; i++) {
        printf("%d", i->id);
    }
}

头结点：

void printList(Node *head) {
    Node *i;
    for (i = head->next; i != NULL; i++) {
        printf("%d", i->id);
    }
}

二、双向链表

节点(Node) 结构体：

struct Node {
    int id;
	Node *prev;
	Node *next;
	/* ... */
}

维护方式同单向链表，但有时为了方便也会增加 尾指针 或 尾节点。

指针：

Node *head = NULL;
Node *tail = NULL;

结点：

Node head;
Node tail;
head.prev = NULL;
head.next = &tail;
tail.prev = &head;
tail.next = NULL;

1. 添加节点

  1. 创建节点

    这一步是在内存中开辟一段空间给节点，并把相应数据存储进去。

    Node *createNode(int id) {
	Node *p = malloc(sizeof(Node));
    p -> id = id;
    return p;
}

  2. 将节点添加到链表中

    头指针：

    void *addNode(Node *p, Node **head, Node **tail) {
    if (*head == NULL)
        *tail = p;
    p->next = *head;
    *head = p;
}

    头结点：

    void addNode(Node *p, Node *head) {
    p->next = head->next;
    p->prev = head;
    head->next->prev = p;
	head->next = p;
}

2. 删除节点

删除节点p

将节点p从链表中移除并释放内存

头指针：

// 删除头结点
void delHeadNode(Node **head, Node **tail) {
    if (*head == NULL) return;
    Node *del = *head;
    
    if (*head == *tail) {
        *head = *tail = NULL;
    	free(del);
        return;
    }
    
    *head = del->next;
    del->next->prev = NULL;
    free(del);
}
// 删除 p 节点后的一个节点
void delNode(Node *p, Node ) {
    if (p->next == NULL) return;
    Node *del = p->next;
    
    p->next = del->next;
    free(del);
}

头结点：

void delNode(Node *p) {
    if (p->next == NULL) return;
    Node *del = p->next;
    
    p->next = del->next;
    free(del);
}

3. 遍历链表

头指针：

void printList(Node *head) {
    Node *i;
    for (i = head; i != NULL; i = i->next) {
        printf("%d", i->id);
    }
}

头结点：

void printList(Node *head) {
    Node *i;
    for (i = head->next; i->next != NULL; i = i->next) {
        printf("%d", i->id);
    }
}
