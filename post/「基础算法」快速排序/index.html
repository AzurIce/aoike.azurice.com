「基础算法」快速排序

板子：

#include <iostream>
#include <cstdio>
#include <cstdlib>

#define MAXN (int)(1E6 + 7)

using namespace std;

void qRead(int &x) {
	char c; x = 0;
    while ((c = getchar()) < '0' || c > '9');
    while ('0' <= c && c <= '9') {
        x = x*10 + c - '0';
        c = getchar();
    }
}

void qSort(int *a, int l, int r) {
    if (l == r) return ;
    int x = a[rand()%(r-l) + l];
    int i = l-1, j = r+1;
    while (i < j) {
        do i++; while (a[i] < x);
        do j--; while (a[j] > x);
        if (i < j) swap(a[i], a[j]);
    }
    qSort(a, l, j);
    qSort(a, j+1, r);
}

int a[MAXN];
int main() {
    int n; scanf("%d", &n);
    for (int i = 1; i <= n; i++) qRead(a[i]);
    qSort(a, 1, n);
    for (int i = 1; i <= n; i++) printf("%d ", a[i]); puts("");
    return 0;
}

几个关键点

  1. int x = a[rand()%(r-l) + l] 中使用 r-l 而非 r-l+1 。

    由于划分为 [l, j] 和 [j+1, r] ，若 x 取到 a[r] 则会使 j == r 无限循环。

  2. 在 do i++ 和 do j-- 处使用 do...while

    为了确保 a[i] == x 或 a[j] == x 时，i 与 j 均会移动。
